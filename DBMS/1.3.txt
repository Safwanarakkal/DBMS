b) DCL COMMAND 
Create two users user1, user2 
1.      GRANT 
->Give insert and select privilege to user 1 on customer table in the invoice database. 
->Give update and delete privilege to user 2 on all tables in the invoice database. 


mysql> create user 'safwans'@'localhost'identified by 'mypass1';
Query OK, 0 rows affected (0.03 sec)

mysql> GRANT INSERT,SELECT ON customer TO 'safwan'@'localhost';

mysql> GRANT DELETE,UPDATE ON invoice.*TO 'safwan'@'localhost';
